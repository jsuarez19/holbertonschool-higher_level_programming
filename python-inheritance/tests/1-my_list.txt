#The ``1-my_list`` module ::
=====================================

#Import ``MyList`` class
----------------------
>>> MyList = __import__('1-my_list').MyList

#Pass list int positive
-----------------------
>>> MyList.print_sorted([1, 4, 2, 3, 5])
[1, 2, 3, 4, 5]

#Pass list int negative
------------------------
>>> my_list = MyList()
>>> my_list.append(-2)
>>> my_list.append(-5)
>>> my_list.append(-4)
>>> my_list.print_sorted()
[-5, -4, -2]

#Pass empty list
----------------------
>>> my_list = MyList()
>>> my_list.print_sorted()
[]

#Check __str__
----------------------
>>> my_list = MyList()
>>> print(my_list)
pass
